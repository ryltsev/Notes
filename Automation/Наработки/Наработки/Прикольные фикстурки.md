1. Получение токена

```
@pytest.fixture(scope="module")
def get_token(url):
    def _get_token(login, password="12345678"):
        session = requests.Session()
        user_data = {"login": login, "password": password}

        response = session.post(url + CredentialsLogin.path_login, data=user_data)

        # Проверяю код ответа
        if response.status_code == 200:
            result_tokentype = response.json()["data"]["access_type"]
            result_accesstoken = response.json()["data"]["access_token"]
            access_token = f"{result_tokentype} {result_accesstoken}"
            return access_token
        elif response.status_code == 401:
            raise requests.exceptions.RequestException("Invalid Credentials")
        return None

    return _get_token
```
